#include <stdlib.h>
#include <time.h>
#include <stdio.h>

int gcd(int a, int b)
{
    return (b ? gcd(b, a % b) : a);
}

void main()
{
    struct timeval now;
    struct tm time;

    gettimeofday(&now, 0);
    localtime_r(&(now.tv_sec), &time);

    time.tm_mon += 1;
    printf("Password: %x%02d-%02x%02d\n", time.tm_mon, time.tm_mon, time.tm_mday, gcd(time.tm_mon, time.tm_mday));
}
